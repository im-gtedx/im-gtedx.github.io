    import Conversation from "./conversation";

/**
 * 会话查询.
 */
class ConversationQuery {

    constructor(imClient) {
        this.imClient = imClient;
    }

    /**
     * 获取当前组织的当前用户的会话列表.
     *
     * @returns {*|PromiseLike<T>|Promise<T>}
     */
    getOrgConversations() {
        let {oid} = this.imClient.config;
        let params = {body: {tenant_id: oid}};
        return this.imClient.api.getOrgConversations(params)
            .then((response) => {
                let {code, message, body} = response.data;
                if (code == 200) {
                    if (body) {
                        let conversationResults = [];
                        for (let index in body) {
                            let conversationSource = body[index];
                            let conversationTarget = new Conversation(this.imClient);
                            let conversation = Object.assign(conversationTarget, conversationSource);
                            conversationResults.push(conversation);
                        }
                        return conversationResults;
                    }
                } else {
                    throw new Error(message);
                }
            });
    }

}

export default ConversationQuery;
