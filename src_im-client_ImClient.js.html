    import EventEmitter from 'eventemitter3';
import ET from '../ImEvent';
import ImConfig from '../ImConfig';
import NetApi from '../net'
import Conversation from "../conversation/conversation";
import ImType from "./ImType";
import ConversationField from "../conversation/conversation-field";
import MessageField from "../messages/message-field";
import Log from "../log";
import TypedMessage from "../messages/typed-message";
import Cache from '../cache/Cache';


class ImClient extends EventEmitter {

    /**
     *
     * @param config
     */
    constructor(config) {
        super();

        if (!(config instanceof ImConfig)) {
            throw new Error('ImClient 实例需要一个ImConfig实例.');
        }

        this.api = new NetApi(config.uid, config.env);
        this.config = config;
        this.cache = new Cache();

        this.on(ET.ConversationRefresh, this._refreshConversations);

        this._createWsConnection();
    }

    /**
     * 获取当前会话列表.
     *
     * @returns {Array<Conversation>} 会话列表
     */
    get currentConversationList() {
        return this._conversations;
    }

    /**
     * 更新当前会话列表.
     *
     * @param {Array<Conversation>}  conversations 会话列表
     */
    set currentConversationList(conversations) {
        this._conversations = conversations;
    }

    /**
     * 设当前ImClient的缓存.
     *
     * @param {Cache} cache 缓存实例
     * @tutorial d007-cache
     */
    setCache(cache) {
        this.cache = cache;
    }


    /**
     * 重新拉取会话.
     * @private
     */
    _refreshConversations = () => {
        this.getConversations();
    }


    /**
     * 创建WebSocket链接.
     * @private
     */
    _createWsConnection() {
        this.ws = new WebSocket(this.config.getWsUrl());

        this.ws.onopen = () => {
            this.pingTimer = setInterval(() => this.ws.ping(), this.config.heartbeats);
        };

        this.ws.onmessage = (event) => {
            if (typeof event.data === 'string') {
                this._emitAndCache(event.data);
            }

            if (event.data instanceof ArrayBuffer) {
                throw new TypeError("接收到的是ArrayBuffer类型的数据");
            }

            if (event.data instanceof Blob) {
                throw new TypeError("接收到的是Blob类型的数据");
            }
        };

        this.ws.onclose = ev => {
            this.emit(ET.Close, ev);
        };

        this.ws.onerror = ev => {
            this.emit(ET.Error, ev);
        };
    }


    /**
     * 创建会话.
     *
     * @param {Object} params {title:"私聊1",remark:"",manager_id:"",icon:"",user_ids:["9896d0783afa55ff194248ecc408baeb"],attrs:"",unique:true}
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    createConversation(params) {
        if (!params) {
            throw new Error("创建会话的参数不能为空");
        }

        if (!params.title) {
            throw new Error("创建会话的名称不能为空");
        }

        if (!Array.isArray(params.user_ids) || params.user_ids.length == 0) {
            throw new Error("创建会话的成员不能为空");
        }

        // 参数最外层：body
        let assignParams = {body: Object.assign(params, {manager_id: this.config.uid})};
        Log.log(JSON.stringify(assignParams));

        let promise = this.api.createConversation(assignParams);
        return promise.then((response) => {
            let {code, message, body} = response.data;
            if (code == 200) { // 每一次创建会话成功之后，必须重新拉取会话列表
                this.emit(ET.ConversationRefresh);
                return Object.assign(new Conversation(this), body); // 赋予发送消息的行为
            } else {
                throw new Error(message);
            }
        });

    }

    /**
     * 根据会话ID获取一个会话实例【待实现】.
     *
     * @param {String} cid 会话ID     *
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    getConversation(cid) {

    }

    /**
     * 获取会话列表.
     */
    getConversations() {

        /**
         * 赋予给Conversation行为方法.
         *
         * @param conversations 会话列表数据
         * @param isWithLastMsg 是否获取当前会话的最后一条消息
         */
        let mapConversations = (conversations, isWithLastMsg) => {
            let conversationResults = [];
            for (let conversationSource of conversations) {
                let conversationTarget = new Conversation(this);
                let conversation = Object.assign(conversationTarget, conversationSource);
                conversationResults.push(conversation);
            }

            if (isWithLastMsg) {
                this.cache.getConversationsWithLastMessage(conversationResults)
                    .then(cb => {
                        if (cb.data) this.currentConversationList = cb.data;
                        this.emit(ET.Conversations, this.currentConversationList);
                    });
            } else {
                this.currentConversationList = conversationResults;
                this.emit(ET.Conversations, this.currentConversationList);
            }
        }


        // 首先, 获取本地数据
        this.cache.getConversations()
            .then(cb => {
                if (cb.data) {
                    mapConversations(cb.data, false);
                } // 其次, 获取远程数据
                return this.api.getConversations();
            })
            .then((response) => {
                let {code, body} = response.data;
                if (code == 200 && body && body.length > 0) {
                    mapConversations(body, true);
                    this.cache.cacheConversations(body);
                }
            })
            .catch((error) => { // todo ...
                this.emit(ET.Error, error.message);
            });
    }

    /**
     * 关闭连接.
     *
     * @returns {Promise<any>}
     */
    close() {
        return new Promise((resolve, reject) => {
            try {
                let currentWsState = this.ws.readyState;
                if (currentWsState == WebSocket.CONNECTING) {
                    reject("正在连接中,不能进行此操作.");
                } else if (currentWsState == WebSocket.OPEN) {
                    clearInterval(this.pingTimer);
                    this.ws.close();
                    resolve();
                } else if (currentWsState == WebSocket.CLOSING) {
                    reject("正在关闭中,不能重复此操作.");
                } else if (currentWsState == WebSocket.CLOSED) {
                    resolve();
                }
            } catch (e) {
                reject(e.message);
            }
        });
    }


    /**
     * @param messageWraper
     * @private
     */
    _emitAndCache(messageWraper) {
        // todo 注意json对象的转换.
        messageWraper = JSON.parse(messageWraper);
        let type = messageWraper.type;
        let message = messageWraper.message;

        switch (type) {
            case ImType.OnLine:
                let postMsg = this._post2EmitMsg(message); // send
                this.cache.cacheMessage(postMsg); // cache single message
                this._emit2Conversations(postMsg); // send to conversation
                break;

            case ImType.OffLine:
                if (!message) return; // send
                let messages = [];
                message.forEach((msg) => {
                    let postMsg = this._post2EmitMsg(msg);
                    messages.push(postMsg);
                });
                this.cache.cacheMessages(messages); // cache multi messages
                this._emit2Conversations(messages);// send to conversation
                break;

            case ImType.Received:
                // this.emit(ET.Message, message);
                // this.cache.cacheMessage(message);
                // this._emit2Conversations(message);
                break;

            case ImType.ErrorMsg:
                this.emit(ET.Error, message);
                break;
        }
    }

    /**
     * post message io and send message.
     *
     * @param message
     * @private
     */
    _post2EmitMsg = (message) => {
        // todo 根据消息类型new 不同类型的实例.
        let msgInstance = Object.assign(new TypedMessage(), message);
        msgInstance[MessageField.Io] = false;
        this.emit(ET.Message, msgInstance);
        return msgInstance;
    }


    /**
     * 将单条或多条消息广播给所属的会话.
     * 为了拦截在单个会话页面的消息接收事件
     *
     * @param message
     * @private
     */
    _emit2Conversations(message) {
        if (message instanceof Array) {
            for (let conversation of this.currentConversationList) {
                let cid = conversation[ConversationField.ConversationId];
                for (let msg of message) {
                    if (cid === msg[MessageField.ConversationID]) {
                        // todo 将消息置为已读
                        conversation.emit(ET.Message, msg);
                    }
                }
            }
        } else {
            for (let conversation of this.currentConversationList) {
                let cid = conversation[ConversationField.ConversationId];
                if (cid === message[MessageField.ConversationID]) {
                    // todo 将消息置为已读
                    conversation.emit(ET.Message, message);
                }
            }
        }
    }

}

export default ImClient;
