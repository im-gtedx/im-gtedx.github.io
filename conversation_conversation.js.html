    import ConversationField from "./conversation-field";
import Message from "../messages/message";
import MessageField from "../messages/message-field";
import EventEmitter from 'eventemitter3';
import ET from "../ImEvent";
import Log from "../log";
import Cb from "../cache/Cb";
import MessageFactory from "../messages/message-factory";

class Conversation extends EventEmitter {


    /**
     * 会话.<br/>
     * 当前对象所有的成员属性请查看 {@link ConversationField}.
     *
     * @extends EventEmitter
     * @param {ImClient} imClient 当前用户实例.
     * @throws {Error} ImClient 实例不能为空.
     */
    constructor(imClient) {
        if (!imClient) throw new Error("ImClient 实例不能为空");
        super();

        this.imClient = imClient;
        this.ws = imClient.ws;
        this.config = imClient.config;
        this.cache = imClient.cache;
        this.api = imClient.api;

        Object.assign(this, {[ConversationField.UnreadCount]: 0});
    }


    /**
     * 在当前的会话中发送一条消息.
     *
     * @param {Message} msg 消息对象:包括其子类.
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    send(msg) {

        if (!(msg instanceof Message)) throw new Error("消息体必须按照格式来");

        let attributes = msg.attributes;
        if (attributes) {
            if (!(typeof attributes === "object")) throw new Error(`自定义属性必须是一个普通的对象`);
            msg[MessageField.Attrs] = JSON.stringify(attributes);
        }

        let message = Object.assign(
            msg,
            {
                [MessageField.ConversationID]: this[ConversationField.ConversationId],
                [MessageField.SendBy]: this.config.id,
                [MessageField.Io]: true
            }
        );

        Log.log(message);

        // todo 发送之前验证 [WebSocket] 连接
        // 首先,缓存
        return this.cache.cacheMessage(message)
            .then((cb) => {
                if (Cb.CodeOk === cb.code) {

                    // 其次,展现 (本地发送)
                    this.emit(ET.Message, message);

                    // 最后,发送到服务端
                    let wsMsg = JSON.stringify(message);

                    try {
                        this.ws.send(wsMsg);
                        return message;
                    } catch (e) {
                        throw new Error(e.message);
                    }
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 获取当前会话的缓存的所有消息.
     *
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    getMessages() {
        let cid = this[ConversationField.ConversationId];
        return this.cache.getMessages(cid)
            .then(cb => {
                if (cb.data) {
                    // 统一处理：给消息赋予行为
                    let messages = [];
                    cb.data.forEach(value => {
                        let msg = MessageFactory.createMessage(value);
                        messages.push(msg);
                    });
                    return messages;
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 获取当前会话的最后一条消息.
     *
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    getLastMessage() {
        let cid = this[ConversationField.ConversationId];
        return this.cache.getLastMessageByCid(cid)
            .then(cb => {
                if (cb.data) {
                    return MessageFactory.createMessage(cb.data);
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 清除当前会话下的所有消息.
     *
     * @returns {PromiseLike<T> | Promise<T> | *}
     */
    clearMessages() {
        return this.cache.clearConversationMessages(this)
            .then(cb => {
                if (Cb.CodeOk === cb.code) {
                    // 发送清空当前会话的消息的事件
                    this.emit(ET.ConversationClearMessages);
                    return this;
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 将当前会话中的未读消息置为已读.
     */
    read() {
        let cid = this[ConversationField.ConversationId];
        return this.cache.readMessageByCid(cid)
            .then(cb => {
                if (Cb.CodeOk === cb.code) {
                    // read之后必须刷新一下会话列表的消息数量
                    return this.imClient.getUnreadMsgCount();
                } else {
                    throw new Error(cb.message);
                }
            })
            .then(() => {
                return this;
            });
    }


    /**
     * 成员 - 自身主动加入[待实现].
     */
    join() {
        // todo
    }


    /**
     * 成员 - 自身主动退出[待实现].
     */
    quit() {
        // todo
    }


    /**
     * 群主 - 添加其他成员.
     *
     * @param {Array<String>} memberIds 成员数组.
     * @returns {PromiseLike<T>}
     */
    add(memberIds) {
        let params = {body: {id: this[ConversationField.ConversationId], user_ids: memberIds}};
        return this.api.addMembers(params)
        // ① 服务端会话更新成功
            .then(response => {
                let {code, message, body} = response.data;
                if (code == 200) { // update db
                    return body;
                } else {
                    throw new Error(message);
                }
            }) // ② 更新数据库中的会话数据
            .then(conversation => {
                return this.cache.cacheConversation(conversation);
            })// ③ 更改当前对象
            .then(cb => {
                if (Cb.CodeOk === cb.code) {
                    let members = this[ConversationField.MemberIds].slice();
                    for (let uid of memberIds) {
                        members.push(uid);
                    }
                    this[ConversationField.MemberIds] = members;
                    return this;
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 群主 - 移除其他成员.
     *
     * @param {Array<String>} memberIds 成员数组.
     * @returns {PromiseLike<T>}
     */
    remove(memberIds) {
        let params = {body: {id: this[ConversationField.ConversationId], user_ids: memberIds}};
        return this.api.removeMembers(params)
        // ① 服务端会话更新成功
            .then(response => {
                let {code, message, body} = response.data;
                if (code == 200) { // update db
                    return body;
                } else {
                    throw new Error(message);
                }
            }) // ② 更新数据库中的会话数据
            .then(conversation => {
                return this.cache.cacheConversation(conversation);
            })// ③ 更改当前对象
            .then(cb => {
                if (Cb.CodeOk === cb.code) {
                    let members = this[ConversationField.MemberIds].slice();
                    for (let uid of memberIds) {
                        let position = members.indexOf(uid);
                        members.splice(position, 1);
                    }
                    this[ConversationField.MemberIds] = members;
                    return this;
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 群主 - 修改群组信息 [ 群标题 群头像 ].
     *
     * @returns {PromiseLike}
     */
    save() {

        let params = {
            body: {
                id: this[ConversationField.ConversationId],
                title: this[ConversationField.Title],
                icon: this[ConversationField.Icon]
            }
        };

        return this.api.modifyGroupInfo(params)
            .then(response => {
                let {code, message, body} = response.data;
                if (code == 200) {
                    return body;
                } else {
                    throw new Error(message);
                }
            })// ② 更新数据库中的会话数据
            .then(conversation => {
                return this.cache.cacheConversation(conversation);
            })// ③ 返回当前对象
            .then(cb => {
                if (Cb.CodeOk === cb.code) {  // 更改当前对象
                    this.imClient.refreshConversations();
                    return this;
                } else {
                    throw new Error(cb.message);
                }
            });
    }


    /**
     * 查询成员数量【待实现】.
     *
     * @returns {number} 会话成员的数量
     */
    count() {
        let uids = this[ConversationField.MemberIds];
        if (uids) return uids.length;
        return 0;
    }

}

export default Conversation;
